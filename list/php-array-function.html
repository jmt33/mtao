php Array function
<?php
header("Content-type:text/html; charset=utf-8");
function pr($vs){
    echo "<pre>";
    print_r($vs);
    echo "</pre>";
}
//一维数组
$arrOne = array('nanjing','qingdao','hefei','jinan','heze','beijing','xuzhou');
//二维数组
$arrTwo = array('appale'=>'100','orange'=>'20','banana'=>'30','pear'=>'50','grape'=>'50');
//获得数组中键值
pr(array_values($arrTwo));
//获得数组键
pr(array_keys($arrTwo));
//数组中的值与键名互换（如果有重复前面的会被后面的覆盖）
pr(array_flip($arrTwo));
//检测数组中是否有某个值
if (in_array('100', $arrTwo)) {
    pr('有100');
}
//在数组中检索 ，如果存在返回键名
pr(array_search('20', $arrTwo));
//检测键名是否存在数组中
if(array_key_exists('orange', $arrTwo)) {
    pr('橘子存在');
}
//检测键名是否存在数组中
if(isset($arrTwo['orange'])) {
    pr('橘子存在');
}
//将数组反转
pr(array_reverse($arrTwo));
//数组内部指针
foreach ($arrTwo as $key => $value) {
    //pr(current($arrTwo));
}
//回调
function fiter($var) {
    if ($var%3 === 0) {
        return true;
    }
}
//打印出过滤后的结果
pr(array_filter($arrTwo,'fiter'));
//多参数回调
function morefiter($v1, $v2) {
    if ($v1 === $v2) {
        return '一样';
    } else {
        return '不一样';
    }
}
$d1 = array('y1','h2','h3','h4');
$d2 = array('y1','y2','y3','y4');
pr(array_map('morefiter', $d1, $d2));

//排序
//由小到大进行排序
$paixu = array('5','8','48','89');
sort($paixu);
pr($paixu);
//由大到小排序
rsort($paixu);
pr($paixu);

//二维数组进行排序  
//根据键名字进行排序
//ksort 由小到大排序
//krsort 由大到小排序
//根据键值进行排序
//asort 由小到大排序
//arsort 由大到小进行排序
/***********下面根据键值进行排序键名会被忽略**********************/
//sort 由小到大
//rsort 由大到小
//根据自定义规则进行排序
usort($arrOne,'sortByLen');
print_r($arrOne);
function sortByLen($one, $two) {
    if (strlen($one) === strlen($two)) {
        return 0;
    } else {
        return (strlen($one) > strlen($two)) ? 1 : -1;
    }
}

//拆分 合并 分解

pr(array_slice($arrOne, 1, 2));
pr(array_splice($arrOne, 1,1));
pr($arrOne);

$num = array('12','45','78','78','45','45');
$bine = array_combine($arrOne, $num);
pr($bine);

pr(array_merge($arrTwo,$bine));

/**
  array_intersect 计算数组交集
  aray_diff 计算数组差集
  array_push 入栈
  array_pop 出栈
  array_shift() 删除数组第一个元素并返回
  array_rand() 从数组中随机选择 1个 或多个元素并返回  该函数有两个参数
  shuffle() 数组随机顺序进行排列
  range(1,6) 声明 1-6 的数组
 **/
?>
